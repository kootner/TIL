    class Solution {
        public int solution(int n) {
            int answer = 0;

              for (int i = 1; i < n+1; i++) {
                if (n % i == 0)
                    answer += i;
            }
            return answer;
        }
    }

    import java.util.Arrays;

    class Solution {
        public int solution(int[] d, int budget) {
            int answer = 0;

            Arrays.sort(d);

             for(int i =0; i< d.length; i++){
                budget -= d[i];
                if(budget>=0)
                    answer++;
            }        
            return answer;
        }
    }
